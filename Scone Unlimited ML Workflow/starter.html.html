Data Staging¶

  1. Extract the data from a hosting service
  2. Transform it into a usable shape and format

import numpy as np
def save_images(df, img_data, train= True):
    #Grab the image data in row-major form
    filenames= df['filenames']
    rows= df['row'] 
    for row, filename in zip(rows, filenames):
      img = img_data[b'data'][row]

      target= np.dstack((
      img[0:1024].reshape(32,32),
      img[1024:2048].reshape(32,32),
      img[2048:].reshape(32,32)))

      if train:
        filepath= f"./train/{filename}"
      else:
        filepath= f"./test/{filename}"
      try:
        plt.imsave(filepath, target)
        print(f"{filename} has been saved successfully")
      except RuntimeError as e:
         return f"Error: {e} \nAn error encountered while saving {filename} file to the target folder located at {filepath}"

def to_metadata_file(df, prefix):
    df["s3_path"] = df["filenames"]
    df["labels"] = df["labels"].apply(lambda x: 0 if x==8 else 1)
    return df[["row", "labels", "s3_path"]].to_csv(
        f"{prefix}.lst", sep="\t", index=False, header=False
    )
